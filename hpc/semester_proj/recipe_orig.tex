\renewcommand{\invariant}{ 
\FlaTwoByOne{ B_T }
            { B_B } 
= 
\FlaTwoByOne{ L_{TL}^{-1} \hat{B}_T }
            { \hat{B}_B } 
}


\renewcommand{\guard}{ \neg \SameSize( L_{TL}, L ) }

$
L \rightarrow
\FlaTwoByTwo{ L_{TL} } { 0 }
              { L_{BL} } { L_{BR}   }
$,
$
B \rightarrow 
\FlaTwoByOne{ B_T }
            { B_B }
$ and
$
\hat{B} \rightarrow 
\FlaTwoByOne{ \hat{B}_T }
            { \hat{B}_B }
$
}


\renewcommand{\partitionsizes}{$ L_{TL} $ is $ 0 \times 0 $, and ... }




\renewcommand{\repartitionings}{

$
\FlaTwoByTwo{ L_{TL} } { 0 }
              { L_{BL} } { L_{BR}   } \rightarrow
\FlaThreeByThreeBR{ L_{00} }{ 0 }{ 0 }
              { l_{10}^T }{ \lambda_{11} }{ 0 }
              { L_{20} }{ l_{21} }{ L_{22} }
$,
$
\FlaTwoByOne{ B_T }
            { B_B }
\rightarrow
\FlaThreeByOneB{ B_0 }
               { b_1^T }
               { B_2 }
$, and \\
$
\FlaTwoByOne{ \hat{B}_T }
            { \hat{B}_B }
\rightarrow
\FlaThreeByOneB{ \hat{B}_0 }
               { \hat{b}_1^T }
               { \hat{B}_2 }
$
}




\renewcommand{\repartitionsizes}{

\renewcommand{\moveboundaries}{ 
\normalsize
$

              { L_{BL} } { L_{BR}   } \leftarrow

              { l_{10}^T }{ \lambda_{11} }{ 0 }
              { L_{20} }{ l_{21} }{ L_{22} }
$,
$
\FlaTwoByOne{ B_T }
            { B_B }
\leftarrow
\FlaThreeByOneT{ B_0 }
               { b_1^T }
               { B_2 }
$, and \\
$
\FlaTwoByOne{ \hat{B}_T }
            { \hat{B}_B }
\leftarrow
\FlaThreeByOneT{ \hat{B}_0 }
               { \hat{b}_1^T }
               { \hat{B}_2 }
$
}




\renewcommand{\beforeupdate}{
\FlaThreeByOneB{ B_0 }
              { b_1^T }
              { B_2 }
=
\FlaThreeByOneB{ L_{00}^{-1} \hat{B}_0 }
              { \hat{b}_1^T }
              { \hat{B}_2 }
}




\renewcommand{\afterupdate}{
\FlaThreeByOneB{ B_0 }
              { b_1^T }
              { B_2 }
=
\FlaThreeByOneB{ L_{00}^{-1} \hat{B}_0 }
              { ( \hat{b}_1^T - l_{10}^T ( L_{00}^{-1} \hat{B}_0 ) ) / \lambda_{11} }
              { \hat{B}_2 }
}




\renewcommand{\update}{
$
\begin{array}{l}
b_1^T \becomes ( b_1^T - l_{10}^T B_0 ) / \lambda_{11} 
\end{array}
$
}
